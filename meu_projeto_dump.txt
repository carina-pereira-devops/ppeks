```yaml
name: "6 - Deploy Backend Java no EKS"

on:
  workflow_dispatch:

permissions:
  id-token: write
  contents: read

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: arn:aws:iam::749000351410:role/github-carina-devops-pipe
          aws-region: us-east-1

      - name: Configure kubectl for EKS
        run: |
          aws eks update-kubeconfig \
            --name ppeks-cluster \
            --region us-east-1

      - name: Ler outputs do RDS via tfstate S3
        id: rds
        run: |
          RDS_HOST=$(aws s3 cp s3://tfstate-ppeks/rds/terraform.tfstate - | \
            python3 -c "import sys,json; s=json.load(sys.stdin); print(s['outputs']['rds_endpoint']['value'])")
          RDS_DB=$(aws s3 cp s3://tfstate-ppeks/rds/terraform.tfstate - | \
            python3 -c "import sys,json; s=json.load(sys.stdin); print(s['outputs']['rds_db_name']['value'])")
          echo "rds_host=$RDS_HOST" >> $GITHUB_OUTPUT
          echo "rds_db=$RDS_DB"     >> $GITHUB_OUTPUT

      - name: Criar namespace se nГЈo existir
        run: |
          kubectl get namespace appback || kubectl create namespace appback

      - name: Criar/atualizar Secret com credenciais do banco
        run: |
          USERNAME=$(aws ssm get-parameter --name "/db/username" --query "Parameter.Value" --output text)
          PASSWORD=$(aws ssm get-parameter --name "/db/password" --with-decryption --query "Parameter.Value" --output text)

          kubectl create secret generic app-db-credentials \
            --from-literal=SPRING_DATASOURCE_USERNAME=$USERNAME \
            --from-literal=SPRING_DATASOURCE_PASSWORD=$PASSWORD \
            --namespace appback \
            --dry-run=client -o yaml | kubectl apply -f -

      - name: Deploy backend com Helm
        run: |
          helm upgrade --install java-back ./charts/java-back \
            --namespace appback \
            --set env.POSTGRES_HOST=${{ steps.rds.outputs.rds_host }} \
            --set env.POSTGRES_DB=${{ steps.rds.outputs.rds_db }}
```

resource "helm_release" "karpenter" {
  name       = "karpenter"
  repository = "oci://public.ecr.aws/karpenter"

  namespace  = "kube-system"

  # Aguarda o pod estar Running antes de continuar o terraform
  wait    = true
  timeout = 300

  values = [
    yamlencode({
      # Service Account com IRSA
      serviceAccount = {
        create = true
        name   = "karpenter"
        annotations = {
          "eks.amazonaws.com/role-arn" = aws_iam_role.karpenter_controller.arn
        }
      }

      settings = {
        clusterName       = var.cluster_name
        clusterEndpoint   = var.cluster_endpoint
        interruptionQueue = aws_sqs_queue.karpenter_interruption.name
      }

      tolerations = [{
        key      = "CriticalAddonsOnly"
        operator = "Exists" # в†ђ Exists, nГЈo Equal вЂ” Г© como o EKS usa
      }]

      affinity = {
        nodeAffinity = {
          requiredDuringSchedulingIgnoredDuringExecution = {
            nodeSelectorTerms = [
              {
                matchExpressions = [
                  {
                    key      = "CriticalAddonsOnly"
                    operator = "Exists"
                  }
                ]
              }
            ]
          }
        }
        # Anti-affinity: distribui os 2 pods do Karpenter em AZs diferentes
        podAntiAffinity = {
          preferredDuringSchedulingIgnoredDuringExecution = [
            {
              weight = 100
              podAffinityTerm = {
                labelSelector = {
                  matchLabels = {
                    "app.kubernetes.io/name" = "karpenter"
                  }
                }
                topologyKey = "kubernetes.io/hostname"
              }
            }
          ]
        }
      }

      # 2 rГ©plicas para HA (uma em cada node do MGN)
      replicas = 2

      controller = {
        resources = {
          requests = {
            cpu    = "100m"
            memory = "256Mi"
          }
          limits = {
            cpu    = "500m"
            memory = "512Mi"
          }
        }
      }

      logLevel = "info"
    })
  ]

  depends_on = [
    aws_iam_role_policy_attachment.karpenter_controller,

### iac_lambda/lambda_function.py
```python
import boto3
import os
import json

def lambda_handler(event, context):
    """
    ppeks Health Check Lambda
    - LГЄ credenciais do SSM Parameter Store
    - Retorna status dos recursos do projeto
    - ExposiГ§ГЈo via Function URL (aplicaГ§ГЈo web)
    """
    ssm = boto3.client('ssm', region_name=os.environ.get('AWS_REGION', 'us-east-1'))
    results = {}

    # LГЄ username do banco (SSM jГЎ usado no projeto)
    try:
        param = ssm.get_parameter(
            Name=os.environ['SSM_PATH_DB_USERNAME'],
            WithDecryption=False
        )
        results['db_user_found'] = True
        results['db_user']       = param['Parameter']['Value']
    except Exception as e:
        results['db_user_found'] = False
        results['db_user_error'] = str(e)

    # LГЄ endpoint do RDS
    results['db_host']    = os.environ.get('DB_HOST', 'not-configured')
    results['project']    = os.environ.get('PROJECT_NAME', 'ppeks')
    results['status']     = 'ok'
    results['message']    = 'ppeks Lambda health check executado com sucesso'

    print(json.dumps(results))

    # Retorno compatГ­vel com Function URL (HTTP response)
    return {
        "statusCode": 200,
        "headers": {
            "Content-Type": "application/json"
        },
        "body": json.dumps(results, indent=2)
    }

```

# Lambda

CriaГ§ГЈo de alguns recursos como uma Lambda, apenas para adiГ§ГЈo de crГ©ditos na conta AWS.

A funГ§ГЈo Lambda `ppeks-health-check` Г© uma aplicaГ§ГЈo web serverless que atua como **ponto central de verificaГ§ГЈo de saГєde** dos recursos da infraestrutura do projeto `ppeks`. Ela Г© exposta publicamente via **Function URL** вЂ” ou seja, gera um endpoint HTTP diretamente, sem necessidade de API Gateway.

Quando acessada (via browser, `curl`, ou pelo EventBridge), a funГ§ГЈo executa trГЄs verificaГ§Гµes em sequГЄncia e retorna um JSON com o resultado.

1.1 - Verifica as credenciais no SSM Parameter Store

```python
param = ssm.get_parameter(
    Name=os.environ['SSM_PATH_DB_USERNAME'],
    WithDecryption=False
)
```

LГЄ o parГўmetro `/db/username` do **AWS Systems Manager Parameter Store** вЂ” o mesmo repositГіrio de credenciais usado pelo backend Java e pelo workflow de deploy. Confirma que:

- A IAM Role da Lambda tem permissГЈo de leitura no SSM

- O parГўmetro existe e estГЎ acessГ­vel

1.2 - LГЄ o endpoint do RDS via variГЎvel de ambiente

```python
results['db_host'] = os.environ.get('DB_HOST', 'not-configured')
```

O `DB_HOST` Г© injetado pelo Terraform no momento do `apply`, lendo o endpoint diretamente do tfstate do RDS via `terraform_remote_state`. Isso garante que a Lambda sempre aponta para o banco correto, sem valores hardcoded.

1.3 - Retorna o status como resposta HTTP

```json
{
  "db_user_found": true,
  "db_user": "matthias",
  "db_host": "ppeks-postgres.xxxxxx.us-east-1.rds.amazonaws.com",
  "project": "ppeks",
  "status": "ok",
  "message": "ppeks Lambda health check executado com sucesso"
}
```

A resposta segue o formato esperado pela **Function URL** (`statusCode`, `headers`, `body`), tornando-a compatГ­vel com qualquer browser ou cliente HTTP.

1.4 - Infraestrutura associada

IAM Role

A Lambda executa com uma IAM Role dedicada (`ppeks-lambda-role`) com apenas duas permissГµes:
- `AWSLambdaBasicExecutionRole` вЂ” escrita de logs no CloudWatch
- `AmazonSSMReadOnlyAccess` вЂ” leitura de parГўmetros no SSM

Nenhuma permissГЈo de escrita, nenhum acesso ao banco diretamente вЂ” **princГ­pio do menor privilГ©gio**.

1.5 - Function URL

```
https://<id>.lambda-url.us-east-1.on.aws/
```

Endpoint HTTP pГєblico gerado automaticamente pela AWS. NГЈo requer API Gateway. MГ©todo permitido: `GET`.

1.6 - EventBridge (agendamento)

Um gatilho automГЎtico executa a Lambda **a cada 5 minutos** via `rate(5 minutes)`. Isso serve para:

- Manter a Lambda "aquecida" (reduz cold start)

- Gerar logs periГіdicos no CloudWatch

- Funcionar como um monitor contГ­nuo da infraestrutura

logging.basicConfig(format='%(levelname)s:%(name)s:%(module)s:%(message)s', level=logging.INFO)

# Set a default external API URL
# Override the default URL if an environment variable is set
app.config['BACKEND_URL'] = 'http://localhost:8080/todos/'
app.config['BACKEND_URL'] = os.getenv('BACKEND_URL', app.config['BACKEND_URL'])

@app.route('/')
def index():

    backend_url = app.config['BACKEND_URL']
    response = requests.get(backend_url)
    
    logging.info("GET %s/todos/",backend_url)
    if response.status_code == 200:
        # Print out the response content
        # print(response.text)
        logging.info("Response: %s", response.text)        
        todos = response.json()
        
    return render_template('index.html', todos=todos)

@app.route('/add', methods=['POST'])
def add():

    if request.method == 'POST':
        new_todo = request.form['todo']
        logging.info("POST  %s/todos/%s",app.config['BACKEND_URL'],new_todo)
        response = requests.post(app.config['BACKEND_URL']+new_todo)
    return redirect(url_for('index'))

@app.route('/delete', methods=['POST'])
def delete():

    if request.method == 'POST':
        delete_todo = request.form['todo']
        logging.info("POST  %s/todos/%s",app.config['BACKEND_URL'],delete_todo)
        print(delete_todo)
    response = requests.delete(app.config['BACKEND_URL']+delete_todo)
    return redirect(url_for('index'))

if __name__ == '__main__':
    app.run(host='0.0.0.0')
    # app.run(debug=True) # doesn't work with auto instrumentation
